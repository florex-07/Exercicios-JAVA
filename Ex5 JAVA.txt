public class Ex5 {
    public static void main(String[] args) {
        Carro bmw = new Carro("BMW", "320i", 2022, 4, false);
        Carro mercedes = new Carro("Mercedes", "C200", 2021, 4, true);
        Carro honda = new Carro("Honda", "Civic", 2020, 4, false);
        Mota yamaha = new Mota("Yamaha", "MT-07", 2023, "Desportiva");

        bmw.modoAcelerar();
        mercedes.modoAcelerar();
        honda.modoAcelerar();
        yamaha.modoAcelerar();

        EstadoVeiculo estado = new EstadoVeiculo();
        estado.mover(bmw);
        estado.parado(yamaha);
    }
}

class Veiculo {
    private String marca;
    private String modelo;
    private int ano;

    public Veiculo(String marca, String modelo, int ano) {
        this.marca = marca;
        this.modelo = modelo;
        this.ano = ano;
    }

    public String getMarca() { return marca; }
    public String getModelo() { return modelo; }
    public int getAno() { return ano; }

    public void modoAcelerar() {
        System.out.println("O veículo está a acelerar.");
    }
}

class Carro extends Veiculo {
    private int portas;
    private boolean descapotavel;

    public Carro(String marca, String modelo, int ano, int portas, boolean descapotavel) {
        super(marca, modelo, ano);
        this.portas = portas;
        this.descapotavel = descapotavel;
    }

    public int getPortas() { return portas; }
    public boolean isDescapotavel() { return descapotavel; }

    @Override
    public void modoAcelerar() {
        System.out.println("O carro " + getMarca() + " acelera suavemente com o pedal.");
    }
}

class Mota extends Veiculo {
    private String tipo;

    public Mota(String marca, String modelo, int ano, String tipo) {
        super(marca, modelo, ano);
        this.tipo = tipo;
    }

    public String getTipo() { return tipo; }

    @Override
    public void modoAcelerar() {
        System.out.println("A mota " + getMarca() + " acelera rapidamente com o punho.");
    }
}

class EstadoVeiculo {
    public void mover(Veiculo v) {
        System.out.println("O " + v.getMarca() + " " + v.getModelo() + " está a andar.");
    }

    public void parado(Veiculo v) {
        System.out.println("O " + v.getMarca() + " " + v.getModelo() + " está parado.");
    }
}
